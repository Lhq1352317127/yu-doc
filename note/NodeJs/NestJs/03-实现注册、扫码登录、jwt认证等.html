Nest.js 实战系列二：实现注册、扫码登录、JWT认证等 ​

一、快速创建模块 ​

使用 nest g resource 命令可快速创建包含 Controller、Service、Module 及 DTO 的 REST API 模块：

bash
nest g resource user  # 快速生成用户模块相关文件
1

该命令会自动创建基础的 CRUD 代码结构，简化开发流程。

二、用户注册 ​

1. 密码加密方案 ​

使用 bcryptjs 对密码进行加盐加密，安装依赖：

bash
npm install bcryptjs
1

核心方法：

  • hashSync(data, salt)：同步加密数据（salt 为盐值，推荐设为 10）
  • compareSync(data, encrypted)：验证数据与加密后数据是否匹配

2. 用户实体设计 ​

typescript
// src/user/entities/user.entity.ts
import { Column, Entity, PrimaryGeneratedColumn, BeforeInsert } from 'typeorm';
import * as bcrypt from 'bcryptjs';

@Entity('user')
export class User {
  @PrimaryGeneratedColumn('uuid')  // 使用 uuid 作为主键
  id: string;

  @Column({ length: 100 })
  username: string;  // 用户名（唯一）

  @Column({ length: 100 })
  nickname: string;  // 昵称

  @Column({ select: false })  // 查询时默认不返回密码
  password: string;  // 加密后的密码

  @Column({ default: '' })
  avatar: string;  // 头像

  @Column({ default: '' })
  email: string;  // 邮箱

  @Column('simple-enum', { 
    enum: ['root', 'author', 'visitor'], 
    default: 'visitor' 
  })
  role: string;  // 角色（root/author/visitor）

  @Column({
    name: 'create_time',
    type: 'timestamp',
    default: () => 'CURRENT_TIMESTAMP'
  })
  createTime: Date;

  @Column({
    name: 'update_time',
    type: 'timestamp',
    default: () => 'CURRENT_TIMESTAMP'
  })
  updateTime: Date;

  // 插入数据前自动加密密码
  @BeforeInsert()
  async encryptPwd() {
    this.password = bcrypt.hashSync(this.password, 10);
  }
}

3. 注册业务逻辑 ​

typescript
// src/user/user.service.ts
import { Injectable, HttpException, HttpStatus } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { User } from './entities/user.entity';
import { CreateUserDto } from './dto/create-user.dto';

@Injectable()
export class UserService {
  constructor(
    @InjectRepository(User)
    private userRepository: Repository<User>,
  ) {}

  async register(createUser: CreateUserDto) {
    const { username } = createUser;
    // 检查用户名是否已存在
    const existUser = await this.userRepository.findOne({
      where: { username }
    });
    if (existUser) {
      throw new HttpException('用户名已存在', HttpStatus.BAD_REQUEST);
    }
    // 创建并保存用户
    const newUser = this.userRepository.create(createUser);
    return this.userRepository.save(newUser);
  }
}

4. 过滤密码返回 ​

方法1：通过实体字段配置 ​

在实体中设置 select: false，查询时默认不返回密码（但 save 方法仍会返回，需配合查询语句手动排除）：

typescript
// user.entity.ts
@Column({ select: false })
password: string;

// 注册后手动查询（不含密码）
async register(createUser: CreateUserDto) {
  // ... 省略其他逻辑
  await this.userRepository.save(newUser);
  return this.userRepository.findOne({ where: { username } });  // 此时不返回密码
}

@Exclude()  // 序列化时排除该字段

// 在控制器中使用拦截器

@UseInterceptors(ClassSerializerInterceptor)
@Post('register')
register(@Body() createUser: CreateUserDto) {
  return this.userService.register(createUser);
}

三、本地登录认证（用户名+密码） ​

1. 安装依赖 ​

bash

npm install @types/passport @types/passport-local -D  # 类型声明
1
2

2. 配置 Passport 本地策略 ​

typescript
// src/auth/local.strategy.ts
import { Injectable, BadRequestException } from '@nestjs/common';
import { PassportStrategy } from '@nestjs/passport';
import { Strategy, IStrategyOptions } from 'passport-local';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { User } from '../user/entities/user.entity';
import { compareSync } from 'bcryptjs';

@Injectable()
export class LocalStrategy extends PassportStrategy(Strategy) {
  constructor(
    @InjectRepository(User)
    private userRepository: Repository<User>,
  ) {
    super({
      usernameField: 'username',  // 对应请求体中的 username 字段
      passwordField: 'password'   // 对应请求体中的 password 字段
    } as IStrategyOptions);
  }

  // 验证逻辑（Passport 自动调用）
  async validate(username: string, password: string) {
    // 查询用户（手动包含密码字段）
    const user = await this.userRepository
      .createQueryBuilder('user')
      .addSelect('user.password')  // 强制查询密码
      .where('user.username = :username', { username })
      .getOne();

    if (!user) {
      throw new BadRequestException('用户名不正确');
    }

    // 验证密码
    if (!compareSync(password, user.password)) {
      throw new BadRequestException('密码错误');
    }

    return user;  // 验证通过后，用户信息会挂载到 req.user
  }
}

3. 登录接口实现 ​

typescript
// src/auth/auth.controller.ts
import { Controller, Post, UseGuards, Body, Req, UseInterceptors, ClassSerializerInterceptor } from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';
import { AuthService } from './auth.service';

@Controller('auth')
export class AuthController {
  constructor(private authService: AuthService) {}

  @UseGuards(AuthGuard('local'))  // 使用本地策略守卫
  @UseInterceptors(ClassSerializerInterceptor)  // 过滤密码
  @Post('login')
  async login(@Req() req) {
    // req.user 为 LocalStrategy 中返回的用户信息
    return this.authService.login(req.user);
  }
}

3. 生成 JWT Token ​

typescript
// src/auth/auth.service.ts
import { Injectable } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { User } from '../user/entities/user.entity';

@Injectable()
export class AuthService {
  constructor(private jwtService: JwtService) {}

  // 生成 token
  createToken(user: Partial<User>) {
    return this.jwtService.sign({
      id: user.id,
      username: user.username,
      role: user.role
    });
  }

  // 登录后返回 token
  async login(user: Partial<User>) {
    const token = this.createToken(user);
    return { token };
  }
}

4. 验证 JWT Token ​

配置 JWT 策略 ​

typescript
// src/auth/jwt.strategy.ts
import { Injectable, UnauthorizedException } from '@nestjs/common';
import { PassportStrategy } from '@nestjs/passport';
import { Strategy, ExtractJwt, StrategyOptions } from 'passport-jwt';
import { ConfigService } from '@nestjs/config';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { User } from '../user/entities/user.entity';

@Injectable()
export class JwtStrategy extends PassportStrategy(Strategy) {
  constructor(
    @InjectRepository(User)
    private userRepository: Repository<User>,
    private configService: ConfigService,
  ) {
    super({
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),  // 从请求头 Bearer 中提取 token
      secretOrKey: configService.get('JWT_SECRET')
    } as StrategyOptions);
  }

  // 验证 token 中的用户信息
  async validate(payload: any) {
    const user = await this.userRepository.findOne({
      where: { id: payload.id }
    });
    if (!user) {
      throw new UnauthorizedException('token 无效');
    }
    return user;  // 验证通过后挂载到 req.user
  }
}

五、微信扫码登录 ​

1. 准备工作 ​

  • 申请微信开放平台账号，创建网站应用，获取 AppID 和 AppSecret
  • 配置授权回调域名（如 www.example.com）

2. 登录流程 ​

  1. 获取授权二维码
    后端重定向到微信授权页面：

    typescript
    // src/auth/auth.controller.ts
import { Get, Res, Headers } from '@nestjs/common';
import * as urlencode from 'urlencode';

@Get('wechatLogin')
async wechatLogin(@Headers() headers, @Res() res) {
  const APPID = process.env.WECHAT_APPID;
  const redirectUri = urlencode('https://www.example.com/callback');  // 回调地址（需编码）
  // 重定向到微信授权页面
  res.redirect(
    `https://open.weixin.qq.com/connect/qrconnect?appid=${APPID}&redirect_uri=${redirectUri}&response_type=code&scope=snsapi_login#wechat_redirect`
  );
}

  2. 通过 code 换取 access_token
    微信回调后会携带 code 参数，后端用 code 换取用户信息：

    typescript
    // src/auth/auth.service.ts
import { HttpService } from '@nestjs/axios';
import { firstValueFrom } from 'rxjs';

async getWechatUserInfo(code: string) {
  const APPID = process.env.WECHAT_APPID;
  const APPSECRET = process.env.WECHAT_APPSECRET;
  // 调用微信接口换取 access_token 和 openid
  const res = await firstValueFrom(
    this.httpService.get(
      `https://api.weixin.qq.com/sns/oauth2/access_token?appid=${APPID}&secret=${APPSECRET}&code=${code}&grant_type=authorization_code`
    )
  );
  const { access_token, openid } = res.data;

  // 用 access_token 和 openid 获取用户信息
  const userRes = await firstValueFrom(
    this.httpService.get(
      `https://api.weixin.qq.com/sns/userinfo?access_token=${access_token}&openid=${openid}&lang=zh_CN`
    )
  );
  return userRes.data;  // 微信用户信息（含 nickname、avatar 等）
}

  3. 登录或注册
    根据 openid 判断用户是否已存在，不存在则创建新用户：

    typescript
    // src/auth/auth.service.ts
async loginWithWechat(code: string) {
  const wechatUser = await this.getWechatUserInfo(code);
  // 查找用户（假设用户表有 openid 字段）
  let user = await this.userRepository.findOne({
    where: { openid: wechatUser.openid }
  });
  // 不存在则注册
  if (!user) {
    user = await this.userService.register({
      username: `wechat_${wechatUser.openid}`,
      password: Math.random().toString(36).slice(-8),  // 随机密码
      nickname: wechatUser.nickname,
      avatar: wechatUser.headimgurl,
      openid: wechatUser.openid
    });
  }
  // 生成 token 并返回
  return this.login(user);
}
