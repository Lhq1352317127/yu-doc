import { Injectable, NestMiddleware } from '@nestjs/common';

// è®¡ç®—æ–‡ä»¶MD5ï¼ˆç”¨äºå‘½åå’Œå»é‡ï¼‰
const getFileMd5 = (file: Express.Multer.File): string => {

    .createHash('md5')
    .update(JSON.stringify(file.buffer))
    .digest('hex');
};

@Injectable()

    // é…ç½®å­˜å‚¨è·¯å¾„å’Œæ–‡ä»¶å
