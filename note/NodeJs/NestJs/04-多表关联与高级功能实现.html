多表关联与高级功能实现 ​

一、数据表关系（TypeORM实现） ​

在Nest.js中，TypeORM通过装饰器定义实体间的关系，常见关系类型包括一对一、一对多、多对多。

1. 一对一关系 ​

适用于两个实体一对一映射的场景（如用户与用户档案）。

实体定义示例：

typescript
// user.entity.ts（用户实体）
import { Entity, PrimaryGeneratedColumn, Column, OneToOne, JoinColumn } from 'typeorm';
import { InfoEntity } from './info.entity';

@Entity('user')
export class UserEntity {
  @PrimaryGeneratedColumn()
  id: number;

  @Column()
  username: string;

  // 一对一关联（用户 -> 档案）
  @OneToOne(() => InfoEntity, info => info.user)
  @JoinColumn()  // 外键在user表中（字段名为infoId）
  info: InfoEntity;
}

// info.entity.ts（档案实体）
import { Entity, PrimaryGeneratedColumn, Column, OneToOne } from 'typeorm';
import { UserEntity } from './user.entity';

@Entity('info')
export class InfoEntity {
  @PrimaryGeneratedColumn()
  id: number;

  @Column()
  idcard: string;

  // 一对一关联（档案 -> 用户）
  @OneToOne(() => UserEntity, user => user.info)
  user: UserEntity;
}

2. 一对多关系 ​

适用于“一”方关联“多”方的场景（如分类与文章）。

实体定义示例：

typescript
// category.entity.ts（分类实体）
import { Entity, PrimaryGeneratedColumn, Column, OneToMany } from 'typeorm';
import { PostEntity } from './post.entity';

@Entity('category')
export class CategoryEntity {
  @PrimaryGeneratedColumn()
  id: number;

  @Column()
  name: string;

  // 一对多关联（分类 -> 多篇文章）
  @OneToMany(() => PostEntity, post => post.category)
  posts: PostEntity[];
}

// post.entity.ts（文章实体）
import { Entity, PrimaryGeneratedColumn, Column, ManyToOne, JoinColumn } from 'typeorm';
import { CategoryEntity } from './category.entity';

@Entity('post')
export class PostEntity {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ length: 50 })
  title: string;

  // 多对一关联（文章 -> 所属分类）
  @ManyToOne(() => CategoryEntity, category => category.posts)
  @JoinColumn({ name: 'category_id' })  // 自定义外键字段名
  category: CategoryEntity;
}

3. 多对多关系 ​

适用于双方均可关联多个对方的场景（如文章与标签）。

实体定义示例：

typescript
// post.entity.ts（文章实体）
import { Entity, PrimaryGeneratedColumn, Column, ManyToMany, JoinTable } from 'typeorm';
import { TagEntity } from './tag.entity';

@Entity('post')
export class PostEntity {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ length: 50 })
  title: string;

  // 多对多关联（文章 -> 多个标签）
  @ManyToMany(() => TagEntity, tag => tag.posts)
  @JoinTable({
    name: 'post_tag',  // 中间表名
    joinColumns: [{ name: 'post_id' }],  // 文章侧外键
    inverseJoinColumns: [{ name: 'tag_id' }]  // 标签侧外键
  })
  tags: TagEntity[];
}

// tag.entity.ts（标签实体）
import { Entity, PrimaryGeneratedColumn, Column, ManyToMany } from 'typeorm';
import { PostEntity } from './post.entity';

@Entity('tag')
export class TagEntity {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ length: 50 })
  name: string;

  // 多对多关联（标签 -> 多篇文章）
  @ManyToMany(() => PostEntity, post => post.tags)
  posts: PostEntity[];
}

二、文章模块实现 ​

以博客系统的文章模块为例，实现包含分类、标签、作者关联的完整功能。

1. 文章实体定义 ​

typescript
// post.entity.ts
import { Entity, Column, PrimaryGeneratedColumn, ManyToOne, ManyToMany, JoinTable, Exclude } from 'typeorm';
import { User } from '../user/entities/user.entity';
import { CategoryEntity } from '../category/entities/category.entity';
import { TagEntity } from '../tag/entities/tag.entity';
import { PostInfoDto } from './dto/post-info.dto';

@Entity('post')
export class PostEntity {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ length: 50 })
  title: string;  // 文章标题

  @Column({ type: 'mediumtext', default: null })
  content: string;  // Markdown内容

  @Column({ type: 'mediumtext', default: null, name: 'content_html' })
  contentHtml: string;  // 转换后的HTML

  @Column({ type: 'text', default: null })
  summary: string;  // 摘要

  @Column({ default: null, name: 'cover_url' })
  coverUrl: string;  // 封面图

  @Column({ type: 'int', default: 0 })
  count: number;  // 阅读量

  @Column({ type: 'int', default: 0, name: 'like_count' })
  likeCount: number;  // 点赞量

  @Column({ type: 'tinyint', default: 0, name: 'is_recommend' })
  isRecommend: number;  // 是否推荐

  @Column('simple-enum', { enum: ['draft', 'publish'] })
  status: string;  // 状态（草稿/发布）

  // 关联作者（一对多）
  @ManyToOne(() => User, user => user.posts)
  author: User;

  // 关联分类（多对一）
  @Exclude()  // 序列化时排除
  @ManyToOne(() => CategoryEntity, category => category.posts)
  @JoinColumn({ name: 'category_id' })
  category: CategoryEntity;

  // 关联标签（多对多）
  @ManyToMany(() => TagEntity, tag => tag.posts)
  @JoinTable({
    name: 'post_tag',
    joinColumns: [{ name: 'post_id' }],
    inverseJoinColumns: [{ name: 'tag_id' }]
  })
  tags: TagEntity[];

  @Column({ type: 'timestamp', name: 'publish_time', default: null })
  publishTime: Date;  // 发布时间

  @Column({ type: 'timestamp', default: () => 'CURRENT_TIMESTAMP' })
  create_time: Date;

  @Column({ type: 'timestamp', default: () => 'CURRENT_TIMESTAMP' })
  update_time: Date;

  // 格式化返回数据
  toResponseObject(): PostInfoDto {
    const response: PostInfoDto = { ...this };
    // 处理分类名称
    if (this.category) response.category = this.category.name;
    // 处理标签名称数组
    if (this.tags?.length) response.tags = this.tags.map(tag => tag.name);
    // 处理作者信息
    if (this.author) {
      response.author = this.author.nickname || this.author.username;
      response.userId = this.author.id;
    }
    return response;
  }
}

步骤2：实现角色守卫

typescript
// src/auth/guards/roles.guard.ts
import { Injectable, CanActivate, ExecutionContext, Reflector } from '@nestjs/common';

@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    // 获取接口所需角色
    const requiredRoles = this.reflector.get<string[]>('roles', context.getHandler());
    if (!requiredRoles) return true;  // 无角色要求则直接通过

    // 获取当前用户
    const { user } = context.switchToHttp().getRequest();
    if (!user) return false;  // 未登录

    // 验证用户角色是否匹配
    return requiredRoles.some(role => user.role === role);
  }
}

// src/posts/posts.controller.ts
import { Controller, Post, Body, Req, UseGuards, ApiOperation, ApiBearerAuth } from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';
import { Roles } from '../auth/decorators/roles.decorator';
import { RolesGuard } from '../auth/guards/roles.guard';
import { PostsService } from './posts.service';
import { CreatePostDto } from './dto/create-post.dto';

@Controller('posts')
export class PostsController {
  constructor(private readonly postsService: PostsService) {}

  @ApiOperation({ summary: '创建文章' })
  @ApiBearerAuth()  // Swagger标记需要token
  @Roles('admin', 'root')  // 仅管理员和超级管理员可访问
  @UseGuards(AuthGuard('jwt'), RolesGuard)  // 先验证token，再验证角色
  @Post()
  async create(@Body() createPostDto: CreatePostDto, @Req() req) {
    return this.postsService.create(req.user, createPostDto);
  }
}

3. 联表查询（QueryBuilder） ​

使用TypeORM的QueryBuilder实现多表关联查询，获取文章列表及关联信息。

typescript
// src/posts/posts.service.ts
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { PostEntity } from './entities/post.entity';
import { PostsRo } from './interfaces/posts.ro';

@Injectable()
export class PostsService {
  constructor(
    @InjectRepository(PostEntity)
    private postsRepository: Repository<PostEntity>,
  ) {}

  async findAll(query): Promise<PostsRo> {
    // 创建查询构建器
    const qb = this.postsRepository
      .createQueryBuilder('post')
      .leftJoinAndSelect('post.category', 'category')  // 关联分类
      .leftJoinAndSelect('post.tags', 'tag')  // 关联标签
      .leftJoinAndSelect('post.author', 'user')  // 关联作者
      .orderBy('post.create_time', 'DESC');  // 按创建时间倒序

    // 分页处理
    const { pageNum = 1, pageSize = 10 } = query;
    qb.limit(pageSize);
    qb.offset(pageSize * (pageNum - 1));

    // 执行查询
    const [posts, total] = await qb.getManyAndCount();

    // 格式化返回数据
    return {
      list: posts.map(post => post.toResponseObject()),
      count: total,
    };
  }
}

4. 阅读量统计 ​

实现文章详情查询时自动增加阅读量。

typescript
// src/posts/posts.service.ts
async findById(id: number) {
  // 查询文章详情
  const post = await this.postsRepository
    .createQueryBuilder('post')
    .leftJoinAndSelect('post.category', 'category')
    .leftJoinAndSelect('post.tags', 'tag')
    .leftJoinAndSelect('post.author', 'user')
    .where('post.id = :id', { id })
    .getOne();

  if (!post) {
    throw new HttpException(`id为${id}的文章不存在`, 404);
  }

  // 阅读量+1
  await this.postsRepository.update(id, { count: post.count + 1 });

  return post.toResponseObject();
}

2. 配置文件上传中间件 ​

typescript
// src/common/middleware/upload.middleware.ts
import { Injectable, NestMiddleware } from '@nestjs/common';
import * as multer from 'multer';
import * as crypto from 'crypto';
import * as fs from 'fs';
import * as path from 'path';

// 计算文件MD5（用于命名和去重）
const getFileMd5 = (file: Express.Multer.File): string => {
  return crypto
    .createHash('md5')
    .update(JSON.stringify(file.buffer))
    .digest('hex');
};

@Injectable()
export class UploadMiddleware implements NestMiddleware {
  use(req: any, res: any, next: () => void) {
    // 配置存储路径和文件名
    const storage = multer.diskStorage({
      destination: (req, file, cb) => {
        const filePath = path.join(__dirname, '../../uploads');
        if (!fs.existsSync(filePath)) {
          fs.mkdirSync(filePath, { recursive: true });
        }
        cb(null, filePath);
      },
      filename: (req, file, cb) => {
        const ext = path.extname(file.originalname);  // 获取文件后缀
        const md5 = getFileMd5(file);  // 计算MD5
        cb(null, `${md5}${ext}`);  // 用MD5+后缀命名
      },
    });

    // 创建上传中间件（单文件上传，字段名为file）
    const upload = multer({ storage }).single('file');
    upload(req, res, next);
  }
}

3. 实现COS上传服务 ​

typescript
// src/common/services/cos.service.ts
import { Injectable } from '@nestjs/common';
import { COS } from 'cos-nodejs-sdk-v5';
import { ConfigService } from '@nestjs/config';
import * as fs from 'fs';
import * as path from 'path';

@Injectable()
export class CosService {
  private cos: COS;
  private bucket: string;
  private region: string;
  private cosPath = 'blog/cover/';  // COS中的存储路径

  constructor(private configService: ConfigService) {
    // 初始化COS客户端
    this.cos = new COS({
      SecretId: this.configService.get('TX_SECRET_ID'),
      SecretKey: this.configService.get('TX_SECRET_KEY'),
    });
    this.bucket = this.configService.get('TX_BUCKET');
    this.region = this.configService.get('TX_REGION');
  }

  // 检查文件是否已存在于COS
  async checkFileExists(filename: string): Promise<boolean> {
    return new Promise((resolve) => {
      this.cos.getBucket(
        {
          Bucket: this.bucket,
          Region: this.region,
          Prefix: this.cosPath + filename,
        },
        (err, data) => {
          if (err) resolve(false);
          resolve(!!(data.Contents && data.Contents.length > 0));
        },
      );
    });
  }

  // 上传文件至COS
  async uploadFile(localPath: string, filename: string): Promise<string> {
    return new Promise((resolve, reject) => {
      this.cos.uploadFile(
        {
          Bucket: this.bucket,
          Region: this.region,
          Key: this.cosPath + filename,  // COS中的完整路径
          FilePath: localPath,  // 本地文件路径
          SliceSize: 1024 * 1024 * 2,  // 超过2MB分块上传
        },
        (err, data) => {
          // 删除本地临时文件
          fs.unlinkSync(localPath);
          if (err) reject(err);
          resolve(`https://${data.Location}`);  // 返回文件URL
        },
      );
    });
  }
}

4. 上传接口实现 ​

typescript
// src/upload/upload.controller.ts
import { Controller, Post, UseMiddleware, UploadedFile, UseInterceptors, ApiOperation } from '@nestjs/common';
import { FileInterceptor } from '@nestjs/platform-express';
import { UploadMiddleware } from '../common/middleware/upload.middleware';
import { CosService } from '../common/services/cos.service';

@Controller('upload')
export class UploadController {
  constructor(private cosService: CosService) {}

  @ApiOperation({ summary: '上传文章封面' })
  @UseMiddleware(UploadMiddleware)  // 使用自定义上传中间件
  @Post('cover')
  async uploadCover(@UploadedFile() file) {
    const { filename, path: localPath } = file;

    // 检查文件是否已存在
    const exists = await this.cosService.checkFileExists(filename);
    if (exists) {
      fs.unlinkSync(localPath);  // 删除本地文件
      return { url: `https://${this.cosService.bucket}.cos.${this.cosService.region}.myqcloud.com/${this.cosService.cosPath}${filename}` };
    }

    // 上传至COS
    const url = await this.cosService.uploadFile(localPath, filename);
    return { url };
  }
}
