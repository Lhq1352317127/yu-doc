1. 动态添加对象属性（响应式处理） ​

Vue 对对象的响应式追踪基于初始化时的属性，直接新增属性不会触发视图更新，需用以下方式：

js
// 方式1：Vue 全局 API
Vue.set(obj, '新属性名', 值)

// 方式2：组件实例内使用 $set
this.$set(this.对象, '新属性名', 值)

// 方式3：对象展开（替换式触发响应式）
this.对象 = { ...this.对象, 新属性名: 值 }

（2）多 Checkbox + 全选联动 ​

结合 watch 监听“全选”，computed 计算子项选中状态：

js
export default {
  data() {
    return {
      allChecked: false, // 全选状态
      items: [{ id: 1, checked: false }, { id: 2, checked: false }] // 子项
    }
  },
  watch: {
    // 全选 → 同步子项
    allChecked(newVal) {
      this.items.forEach(item => {
        item.checked = newVal
      })
    },
    // 子项 → 计算全选（深度监听数组对象）
    items: {
      handler() {
        this.allChecked = this.items.every(item => item.checked)
      },
      deep: true
    }
  }
}

4. 数组/对象数组去重 ​

（1）普通数组去重 ​
js
// 方式1：双重循环（兼容旧环境）
function uniqueArr(arr) {
  for (let i = 0; i < arr.length; i++) {
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[i] === arr[j]) {
        arr.splice(j, 1)
        j--; // 修正索引
      }
    }
  }
  return arr;
}

// 方式2：Set 去重（简洁高效，需 ES6+）
const uniqueArr = arr => [...new Set(arr)];

（2）对象数组去重（按指定 key） ​
js
/**
 * 对象数组去重（根据 key 字段）
 * @param {Array} arr - 源数组
 * @param {string} key - 去重依据的字段名
 */
function uniqueObjArr(arr, key) {
  const keyMap = new Map();
  return arr.filter(item => {
    const itemKey = item[key];
    if (!keyMap.has(itemKey)) {
      keyMap.set(itemKey, true);
      return true;
    }
    return false;
  });
}

// 示例：按 id 去重
const uniqueList = uniqueObjArr([{ id: 1, name: 'A' }, { id: 1, name: 'B' }], 'id');

（2）下载逻辑封装 ​
js
/**
 * 下载文件
 * @param {ArrayBuffer} fileBuffer - 文件二进制流
 * @param {string} fileName - 保存的文件名
 * @param {string} fileType - 文件后缀（如 'xlsx' 'pdf'）
 */
function downloadFile(fileBuffer, fileName, fileType) {
  const blob = new Blob([fileBuffer], { type: `application/${fileType}` });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `${fileName}.${fileType}`;
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url); // 释放 blob 资源
}

2. 富文本编辑器（WangEditor 封装示例） ​

在 Vue 组件中封装 WangEditor，支持自定义配置、图片上传：

vue
<template>
  <div ref="editorContainer" style="height: 400px;"></div>
</template>

<script>
import WangEditor from 'wangeditor';

export default {
  data() {
    return {
      editor: null
    };
  },
  mounted() {
    this.initEditor();
  },
  methods: {
    initEditor() {
      this.editor = new WangEditor(this.$refs.editorContainer);
      // 配置图片上传接口
      this.editor.config.uploadImgServer = '/api/upload-image';
      // 自定义颜色、表情等（可选）
      this.editor.config.colors = ['#000', '#ff0000'];
      // 监听内容变化，向外传值
      this.editor.config.onchange = (html) => {
        this.$emit('content-change', html);
      };
      this.editor.create(); // 初始化编辑器
    }
  },
  beforeDestroy() {
    if (this.editor) {
      this.editor.destroy(); // 销毁编辑器，释放资源
    }
  }
};
</script>

3. Vue 项目 SEO 优化 ​

单页应用（SPA）对 SEO 不友好，可通过以下方式优化：

  • 服务端渲染（SSR）：使用 Nuxt.js 等框架，在服务端生成 HTML。
  • 预渲染（Prerendering）：对静态路由生成预渲染 HTML（适合纯静态页）：
    bash
    npm install prerender-spa-plugin -D
    1
    配置 vue.config.js：
    js
    const PrerenderSPAPlugin = require('prerender-spa-plugin');
module.exports = {
  configureWebpack: {
    plugins: [
      new PrerenderSPAPlugin({
        staticDir: path.join(__dirname, 'dist'),
        routes: ['/', '/about'] // 需要预渲染的路由
      })
    ]
  }
};

2. NPM 镜像加速（国内环境） ​

国内访问 npm 官方镜像速度慢，推荐使用淘宝镜像（npmmirror），配置方式如下：

方式1：直接修改 npm 镜像（推荐） ​
bash
# 1. 设置淘宝镜像
npm config set registry https://registry.npmmirror.com/

# 2. 验证镜像（返回淘宝地址则成功）
npm config get registry

# 3. 恢复官方镜像（如需）
npm config set registry https://registry.npmjs.org/
